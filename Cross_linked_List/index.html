十字交叉链表

简介

十字交叉链表（Cross-linked List）是一种用于解决二维结构中交叉关联的数据结构。

c++实现

定义两个链表结构，一个用于行链表，另一个用于列链表。每个节点将包含一个数据元素，以及指向同一行和同一列下一个节点的指针。

函数实现插入和查找操作的实现

  1. 插入节点：
    • insert(int row, int col, int data) 函数用于插入一个新节点。它首先创建一个新节点，并将数据元素赋值给新节点的data成员。然后，它在行链表中查找相应的行，如果找到该行，则调用insertNodeInRow()函数在行链表中插入节点；如果未找到该行，则创建一个新的行链表头节点，并插入新的行链表节点。接下来，它在列链表中查找相应的列，如果找到该列，则调用insertNodeInCol()函数在列链表中插入节点；如果未找到该列，则创建一个新的列链表头节点，并插入新的列链表节点。
  2. 在行链表中插入节点：
    • insertNodeInRow(RowNode* rowPtr, Node* newNode) 函数用于在行链表中插入节点。它首先根据节点的列值找到合适的插入位置，并保持行链表的有序性。如果当前行链表为空，则将新节点直接设置为该行链表的头节点。如果当前列已存在节点，则更新该节点的数据。如果在当前列之后插入节点，则调整指针以插入新节点；如果在当前列之前插入节点，则将新节点插入到该位置。
  3. 在列链表中插入节点：
    • insertNodeInCol(ColNode* colPtr, Node* newNode) 函数用于在列链表中插入节点。它首先根据节点的行值找到合适的插入位置，并保持列链表的有序性。如果当前列链表为空，则将新节点直接设置为该列链表的头节点。如果当前行已存在节点，则更新该节点的数据。如果在当前行之后插入节点，则调整指针以插入新节点；如果在当前行之前插入节点，则将新节点插入到该位置。
  4. 查找节点：
    • findRow(int row) 函数用于在行链表中查找指定行的节点。它通过遍历行链表，逐个比较节点的行值，找到与目标行匹配的节点并返回。如果找不到匹配的节点，则返回nullptr。
    • findCol(int col) 函数用于在列链表中查找指定列的节点。它通过遍历列链表，逐个比较节点的列值，找到与目标列匹配的节点并返回。如果找不到匹配的节点，则返回nullptr。
// 十字交叉链表的节点结构
struct Node {
    int data;
    Node* right;  // 指向同一行下一个节点
    Node* down;   // 指向同一列下一个节点
};

// 十字交叉链表的行链表结构
struct RowNode {
    int row;
    Node* right;  // 指向同一行下一个节点
};

// 十字交叉链表的列链表结构
struct ColNode {
    int col;
    Node* down;   // 指向同一列下一个节点
};

// 十字交叉链表类
class CrossLinkedList {
private:
    RowNode* rows;  // 行链表的头节点
    ColNode* cols;  // 列链表的头节点
public:
    // 构造函数
    CrossLinkedList() {
        rows = nullptr;
        cols = nullptr;
    }
// 插入节点
void insert(int row, int col, int data) {
    // 创建新节点
    Node* newNode = new Node;
    newNode->data = data;
    newNode->right = nullptr;
    newNode->down = nullptr;

    // 在行链表中插入节点
    RowNode* rowPtr = findRow(row);
    if (rowPtr == nullptr) {
        // 行链表中不存在该行的节点
        RowNode* newRow = new RowNode;
        newRow->row = row;
        newRow->right = newNode;
        newRow->down = nullptr;
        insertRowNode(newRow);
    } else {
        // 行链表中存在该行的节点
        insertNodeInRow(rowPtr, newNode);
    }

    // 在列链表中插入节点
    ColNode* colPtr = findCol(col);
    if (colPtr == nullptr) {
        // 列链表中不存在该列的节点
        ColNode* newCol = new ColNode;
        newCol->col = col;
        newCol->down = newNode;
        newCol->right = nullptr;
        insertColNode(newCol);
    } else {
        // 列链表中存在该列的节点
        insertNodeInCol(colPtr, newNode);
    }
}

// 在行链表中插入节点
void insertNodeInRow(RowNode* rowPtr, Node* newNode) {
    Node* curr = rowPtr->right;
    Node* prev = nullptr;

    while (curr != nullptr && curr->right != nullptr && curr->right->col < newNode->col) {
        prev = curr;
        curr = curr->right;
    }

    if (curr == nullptr) {
        // 当前行链表为空
        rowPtr->right = newNode;
    } else if (curr->col == newNode->col) {
        // 当前列已存在节点，更新节点数据
        curr->data = newNode->data;
        delete newNode;
    } else if (curr->col < newNode->col) {
        // 在当前列之后插入节点
        newNode->right = curr->right;
        curr->right = newNode;
    } else if (prev == nullptr) {
        // 在当前列之前插入节点，成为新的头节点
        newNode->right = curr;
        rowPtr->right = newNode;
    } else {
        // 在当前列之前插入节点
        newNode->right = curr;
        prev->right = newNode;
    }
}

// 在列链表中插入节点
void insertNodeInCol(ColNode* colPtr, Node* newNode) {
    Node* curr = colPtr->down;
    Node* prev = nullptr;

    while (curr != nullptr && curr->down != nullptr && curr->down->row < newNode->row) {
        prev = curr;
        curr = curr->down;
    }

    if (curr == nullptr) {
        // 当前列链表为空
        colPtr->down = newNode;
    } else if (curr->row == newNode->row) {
        // 当前行已存在节点，更新节点数据
        curr->data = newNode->data;
        delete newNode;
    } else if (curr->row < newNode->row) {
        // 在当前行之后插入节点
        newNode->down = curr->down;
        curr->down = newNode;
    } else if (prev == nullptr) {
        // 在当前行之前插入节点，成为新的头节点
        newNode->down = curr;
        colPtr->down = newNode;
    } else {
        // 在当前行之前插入节点
        newNode->down = curr;
        prev->down = newNode;
    }
}

// 在行链表中插入新的头节点
void insertRowNode(RowNode* newNode) {
    if (rows == nullptr) {
        rows = newNode;
    } else if (newNode->row < rows->row) {
        newNode->right = rows;
        rows = newNode;
    } else {
        RowNode* curr = rows;
        RowNode* prev = nullptr;

        while (curr != nullptr && curr->row < newNode->row) {
            prev = curr;
            curr = curr->right;
        }

        if (curr == nullptr) {
            prev->right = newNode;
        } else if (curr->row == newNode->row) {
            delete newNode;
        } else {
            newNode->right = curr;
            prev->right = newNode;
        }
    }
}

// 在列链表中插入新的头节点
void insertColNode(ColNode* newNode) {
    if (cols == nullptr) {
        cols = newNode;
    } else if (newNode->col < cols->col) {
        newNode->down = cols;
        cols = newNode;
    } else {
        ColNode* curr = cols;
        ColNode* prev = nullptr;

        while (curr != nullptr && curr->col < newNode->col) {
            prev = curr;
            curr = curr->down;
        }

        if (curr == nullptr) {
            prev->down = newNode;
        } else if (curr->col == newNode->col) {
            delete newNode;
        } else {
            newNode->down = curr;
            prev->down = newNode;
        }
    }
}

// 在行链表中查找节点
RowNode* findRow(int row) {
    RowNode* curr = rows;

    while (curr != nullptr && curr->row != row) {
        curr = curr->right;
    }

    return curr;
}

// 在列链表中查找节点
ColNode* findCol(int col) {
    ColNode* curr = cols;

    while (curr != nullptr && curr->col != col) {
        curr = curr->down;
    }

    return curr;
}
};

应用

应用于具有交叉关系的数据：

  1. 稀疏矩阵：稀疏矩阵是指在矩阵中大部分元素为零的情况。在稀疏矩阵中，非零元素在行和列之间具有交叉关系，因为它们不在同一行或同一列中。
  2. 地图：地图是一个二维空间中的表示，其中包含各种地理信息，例如道路、建筑物、地形等。在地图数据中，各种要素之间存在交叉关系，例如道路可能穿过交叉口，建筑物可能位于交叉点的交叉角上。
  3. 电子表格：电子表格是一种用于组织和管理数据的应用程序。在电子表格中，数据通常按行和列进行排列，并且可以在不同的单元格之间具有交叉关系。公式和函数的引用也可以导致数据之间的交叉关系。
  4. 交叉索引：在数据库或文档中，交叉索引是指通过多个关键字或标签对数据进行组织和检索的方法。这些关键字或标签之间可能存在交叉关系，因为一个数据可以与多个关键字或标签相关联。

任何涉及到数据在不同维度或层级之间相互关联的情况都可以被视为具有交叉关系的数据。
