class Timer:  #@save
    """记录多次运行时间"""
    def __init__(self):
        self.times = []
        self.start()

    def start(self):
        """启动计时器"""
        self.tik = time.time()

    def stop(self):
        """停止计时器并将时间记录在列表中"""
        self.times.append(time.time() - self.tik)
        return self.times[-1]

    def avg(self):
        """返回平均时间"""
        return sum(self.times) / len(self.times)

    def sum(self):
        """返回时间总和"""
        return sum(self.times)

    def cumsum(self):
        """返回累计时间"""
        return np.array(self.times).cumsum().tolist()
        
 
 timer = Timer()
 f'{timer.stop():.5f} sec'

def sgd(params, lr, batch_size):  #@save
    """小批量随机梯度下降"""
    with torch.no_grad():
        for param in params:
            param -= lr * param.grad / batch_size
            param.grad.zero_()
torch.no_grad() 创建一个上下文管理器，该管理器中的计算操作不会被记录用于自动求导。
对于参数列表中的每个参数 param，执行以下操作：
param -= lr * param.grad / batch_size：使用梯度下降更新参数。根据梯度 param.grad、学习率 lr 和批量大小 batch_size，更新参数 param 的值。
param.grad.zero_()：将参数的梯度清零，以便在下一次计算梯度时不受之前梯度的影响。
